  "-" * half + head + "-" * half

    Gets a random article from lurkmore.to

    # File lib/lurkmore.rb, line 15
def self.random
  spinner = TTY::Spinner.new
  spinner.run do
    @session = Capybara::Session.new(:poltergeist)
    @session.visit(LINK_RANDOM)
  end
  begin
    if $stdout.tty?
      puts heading(colorcode(@session.first('.firstHeading').text.strip, GREEN))
    else
      puts heading(@session.first('.firstHeading').text.strip)
    end
    puts
    puts hr
    paragraphs = all_p
    paragraphs.each do |p|
      text = p.text.strip
      next if text.empty? || (text.split(' ').size < 10)
      puts wrap(text), hr
    end
  rescue Errno::EPIPE
    exit(74)
  end
end

def self.wrap(s, width=TERM_WIDTH)
  s.gsub(/(.{1,#{width}})(\s+|\Z)/, "\\1\n")
end
